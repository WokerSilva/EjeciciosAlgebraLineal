\end{equation*}

    \beta' =  \{ (1,3,2), (0,1,2), (1,0,1) \}
\end{equation*}
\begin{equation*}
    \gamma =  \{ (1,1), (2,1) \}
\end{equation*}
\begin{equation*}
    \gamma' =  \{ (3,1), (1,4) \}
\end{equation*}

\begin{enumerate}
    \item Encuentra las matrices de cambios de coordenadas

    \noindent \solucion \\

    \noindent \solucion \\
\end{enumerate}
